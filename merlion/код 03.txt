Option Explicit

' --- Глобальные константы MERLION ---
Private Const MERLION_SHEET_NAME As String = "1 (2)"  ' <<--- УБЕДИТЕСЬ, ЧТО ИМЯ ПРАВИЛЬНОЕ!
Private Const MERLION_INPUT_ARTICUL_COL As String = "P"

' --- УПРАВЛЕНИЕ СТРОКАМИ ---
Private Const MERLION_HEADERS_ROW As Long = 3
Private Const MERLION_DATA_START_ROW As Long = 4


Private Const MERLION_COL_API_ARTICUL As String = "AH"
Private Const MERLION_COL_API_NAME As String = "AI"
Private Const MERLION_COL_API_STOCK As String = "AJ"
Private Const MERLION_COL_API_PRICE_USD As String = "AL"
Private Const MERLION_COL_API_PRICE_RUB As String = "AM"

Private Const MERLION_ENDPOINT_URL As String = "https://apitest.merlion.com/dl/mlservice2"

Private Const MERLION_LOGIN As String = "YOUR_MERLION_CLIENT_ID|API"
Private Const MERLION_PASSWORD As String = "YOUR_MERLION_PASSWORD"

Private Const MERLION_SHIPMENT_METHOD As String = "САМОВЫВОЗ"

Private Const SOAPENV_NS_MERLION As String = "xmlns:soapenv='http://schemas.xmlsoap.org/soap/envelope/'"
Private Const MER_NS_ALIAS_MERLION As String = "mer"
Private Const MER_NS_MERLION As String = "xmlns:" & MER_NS_ALIAS_MERLION & "='https://apitest.merlion.com/dl/mlservice2'"


Public Function IsArrayInitialized(arr As Variant) As Boolean
    On Error Resume Next
    IsArrayInitialized = (LBound(arr) <= UBound(arr))
    On Error GoTo 0
End Function

Sub ProcessMerlionDataFromAPI()
    Dim ws As Worksheet
    Dim http As Object
    Dim lastDataRowInputArticuls As Long
    Dim currentRow As Long
    Dim i As Long
    Dim totalProcessedArticuls As Long
    Dim totalValidArticulsToProcess As Long

    Dim itemApiDetails As Object
    Dim availApiDetails As Object
    Dim articulForApi As String
    Dim originalArticulFromP As String

    On Error GoTo GlobalErrorHandler_Merlion
    currentRow = 0

    Debug.Print "--- Запуск ProcessMerlionDataFromAPI (Merlion mlservice2) ---"
    Debug.Print "Используемый URL эндпоинта: " & MERLION_ENDPOINT_URL
    Debug.Print "Используемый неймспейс Merlion: " & MER_NS_MERLION
    Debug.Print "Строка для ЗАГОЛОВКОВ Merlion: " & MERLION_HEADERS_ROW
    Debug.Print "Начальная СТРОКА ДЛЯ ДАННЫХ (чтение из P и запись в AH-AM): " & MERLION_DATA_START_ROW


    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual
    Application.EnableEvents = False
    Application.StatusBar = "Merlion: Инициализация..."

    Set ws = ThisWorkbook.Sheets(MERLION_SHEET_NAME)

    If MERLION_HEADERS_ROW > 0 And MERLION_HEADERS_ROW < MERLION_DATA_START_ROW Then
        Debug.Print "Merlion: Установка/перезапись заголовков в строке " & MERLION_HEADERS_ROW
        ws.Cells(MERLION_HEADERS_ROW, MERLION_COL_API_ARTICUL).Value = "Артикул"
        ws.Cells(MERLION_HEADERS_ROW, MERLION_COL_API_NAME).Value = "Наименование"
        ws.Cells(MERLION_HEADERS_ROW, MERLION_COL_API_STOCK).Value = "Наличие"
        ws.Cells(MERLION_HEADERS_ROW, "AK").Value = ""
        ws.Cells(MERLION_HEADERS_ROW, MERLION_COL_API_PRICE_USD).Value = "Стоимость в долларах"
        ws.Cells(MERLION_HEADERS_ROW, MERLION_COL_API_PRICE_RUB).Value = "стоимость в рублях"
        ws.Cells(MERLION_HEADERS_ROW, "AN").Value = "стомость в долларах"
    Else
        Debug.Print "Merlion: Пропуск установки заголовков (MERLION_HEADERS_ROW (" & MERLION_HEADERS_ROW & ") некорректна или не меньше MERLION_DATA_START_ROW (" & MERLION_DATA_START_ROW & "))"
    End If
    
    lastDataRowInputArticuls = ws.Cells(Rows.Count, MERLION_INPUT_ARTICUL_COL).End(xlUp).Row

    If lastDataRowInputArticuls < MERLION_DATA_START_ROW Then
        MsgBox "Merlion: Нет артикулов для обработки в столбце " & MERLION_INPUT_ARTICUL_COL & _
               " (начиная со строки " & MERLION_DATA_START_ROW & ").", vbInformation
        GoTo CleanUp_Merlion
    End If

    Dim lastPopulatedRowInOutput As Long
    lastPopulatedRowInOutput = ws.Cells(Rows.Count, MERLION_COL_API_ARTICUL).End(xlUp).Row
    
    Dim rowToClearFrom As Long
    rowToClearFrom = MERLION_DATA_START_ROW
    Dim rowToClearTo As Long
    rowToClearTo = Application.WorksheetFunction.Max(lastDataRowInputArticuls, lastPopulatedRowInOutput)
    
    If rowToClearTo < rowToClearFrom Then
        rowToClearTo = rowToClearFrom - 1
    End If

    If rowToClearTo >= rowToClearFrom Then
        Debug.Print "Merlion: Очистка предыдущих данных в столбцах " & MERLION_COL_API_ARTICUL & "-" & MERLION_COL_API_PRICE_RUB & _
                    " со строки " & rowToClearFrom & " до " & rowToClearTo
        ws.Range(MERLION_COL_API_ARTICUL & rowToClearFrom & ":" & MERLION_COL_API_PRICE_RUB & rowToClearTo).ClearContents
    Else
        Debug.Print "Merlion: Нет старых данных для очистки в диапазоне результатов API (AH-AM, начиная с " & MERLION_DATA_START_ROW & ")."
    End If

    Set itemApiDetails = CreateObject("Scripting.Dictionary")
    itemApiDetails.CompareMode = vbTextCompare
    Set availApiDetails = CreateObject("Scripting.Dictionary")
    availApiDetails.CompareMode = vbTextCompare

    Set http = CreateObject("WinHttp.WinHttpRequest.5.1")
    
    totalValidArticulsToProcess = 0
    For currentRow = MERLION_DATA_START_ROW To lastDataRowInputArticuls
        originalArticulFromP = Trim(CStr(ws.Cells(currentRow, MERLION_INPUT_ARTICUL_COL).Value))
        If originalArticulFromP <> "" And Not IsError(ws.Cells(currentRow, MERLION_INPUT_ARTICUL_COL).Value) And InStr(1, UCase(originalArticulFromP), "ERROR ") = 0 Then
            totalValidArticulsToProcess = totalValidArticulsToProcess + 1
        End If
    Next currentRow
    Debug.Print "Merlion: Всего ВАЛИДНЫХ артикулов для обработки: " & totalValidArticulsToProcess
    If totalValidArticulsToProcess = 0 Then
        MsgBox "Merlion: Нет валидных артикулов для обработки (начиная со строки " & MERLION_DATA_START_ROW & ").", vbInformation
        GoTo CleanUp_Merlion
    End If

    totalProcessedArticuls = 0
    For currentRow = MERLION_DATA_START_ROW To lastDataRowInputArticuls
        originalArticulFromP = Trim(CStr(ws.Cells(currentRow, MERLION_INPUT_ARTICUL_COL).Value))
        articulForApi = originalArticulFromP

        If UCase(originalArticulFromP) = "ZL.BAGEE.003" Then
            articulForApi = "1724795"
            Debug.Print "Merlion: ЗАМЕНА P/N " & originalArticulFromP & " НА ID " & articulForApi & " ДЛЯ ТЕСТА (строка Excel: " & currentRow & ")"
        End If
        
        If originalArticulFromP <> "" And _
           Not IsError(ws.Cells(currentRow, MERLION_INPUT_ARTICUL_COL).Value) And _
           InStr(1, UCase(originalArticulFromP), "ERROR ") = 0 Then

            Debug.Print "Merlion: Обработка P/N: " & originalArticulFromP & ", API запрос с: " & articulForApi & " (строка Excel: " & currentRow & ")"

            Application.StatusBar = "Merlion: getItems для " & articulForApi & " (" & (totalProcessedArticuls + 1) & "/" & totalValidArticulsToProcess & ")"
            Call Merlion_GetItem_API(articulForApi, itemApiDetails, http)
            DoEvents

            Application.StatusBar = "Merlion: getItemsAvail для " & articulForApi & " (" & (totalProcessedArticuls + 1) & "/" & totalValidArticulsToProcess & ")"
            Call Merlion_GetItemAvail_API(articulForApi, availApiDetails, http)
            DoEvents
            
            Application.Wait (Now + TimeValue("00:00:01"))

            ws.Cells(currentRow, MERLION_COL_API_ARTICUL).Value = originalArticulFromP
            Dim itemName As String
            Dim itemStock As Variant
            Dim itemPriceUSD As Variant
            Dim itemPriceRUB As Variant
            Dim tempDictItem As Object
            Dim tempDictAvail As Object

            itemName = "Товар не найден (инфо)"
            itemStock = vbNullString
            itemPriceUSD = vbNullString
            itemPriceRUB = vbNullString

            Debug.Print "Строка Excel: " & currentRow & ", Поиск в словарях по ключу API: '" & articulForApi & "'"

            If itemApiDetails.Exists(articulForApi) Then
                Debug.Print "  Найден в itemApiDetails. Пытаемся получить Name..."
                Set tempDictItem = itemApiDetails(articulForApi)
                If TypeName(tempDictItem) = "Dictionary" Then
                    If tempDictItem.Exists("Name") Then
                        itemName = tempDictItem("Name")
                        Debug.Print "    Name: '" & itemName & "'"
                    Else
                        Debug.Print "    Ключ 'Name' ОТСУТСТВУЕТ в itemApiDetails для '" & articulForApi & "'"
                        itemName = "Имя не найдено в словаре"
                    End If
                Else
                    Debug.Print "    Элемент для '" & articulForApi & "' в itemApiDetails НЕ ЯВЛЯЕТСЯ СЛОВАРЕМ. Тип: " & TypeName(tempDictItem)
                    itemName = "Ошибка данных (item)"
                End If

                If availApiDetails.Exists(articulForApi) Then
                    Debug.Print "  Найден в availApiDetails. Пытаемся получить Stock, PriceUSD, PriceRUB..."
                    Set tempDictAvail = availApiDetails(articulForApi)
                    If TypeName(tempDictAvail) = "Dictionary" Then
                        If tempDictAvail.Exists("Stock") Then
                            itemStock = tempDictAvail("Stock")
                            Debug.Print "    Stock: '" & itemStock & "'"
                        Else
                             Debug.Print "    Ключ 'Stock' ОТСУТСТВУЕТ в availApiDetails для '" & articulForApi & "'"
                        End If
                        
                        If tempDictAvail.Exists("PriceUSD") Then
                            Dim priceUsdVal As Variant: priceUsdVal = tempDictAvail("PriceUSD")
                            Debug.Print "    PriceUSD (сырое): '" & priceUsdVal & "'"
                            Dim preparedPriceUsdStr As String: preparedPriceUsdStr = Replace(CStr(priceUsdVal), ".", Application.DecimalSeparator)
                            If IsNumeric(preparedPriceUsdStr) Then
                                Dim numericPriceUSD As Double: numericPriceUSD = CDbl(preparedPriceUsdStr)
                                If numericPriceUSD > 0 Then
                                    itemPriceUSD = numericPriceUSD
                                    Debug.Print "      PriceUSD (число): " & itemPriceUSD
                                Else: Debug.Print "      PriceUSD (" & numericPriceUSD & ") <= 0"
                                End If
                            Else: Debug.Print "      PriceUSD ('" & priceUsdVal & "' -> '" & preparedPriceUsdStr & "') не является числом после подготовки"
                            End If
                        Else: Debug.Print "    Ключ 'PriceUSD' ОТСУТСТВУЕТ в availApiDetails для '" & articulForApi & "'"
                        End If

                        If tempDictAvail.Exists("PriceClientRUB") Then
                            Dim priceRubVal As Variant: priceRubVal = tempDictAvail("PriceClientRUB")
                            Debug.Print "    PriceClientRUB (сырое): '" & priceRubVal & "'"
                            Dim preparedPriceRubStr As String: preparedPriceRubStr = Replace(CStr(priceRubVal), ".", Application.DecimalSeparator)
                            If IsNumeric(preparedPriceRubStr) Then
                                Dim numericPriceRUB As Double: numericPriceRUB = CDbl(preparedPriceRubStr)
                                If numericPriceRUB > 0 Then
                                    itemPriceRUB = numericPriceRUB
                                     Debug.Print "      PriceClientRUB (число): " & itemPriceRUB
                                Else: Debug.Print "      PriceClientRUB (" & numericPriceRUB & ") <= 0"
                                End If
                            Else: Debug.Print "      PriceClientRUB ('" & priceRubVal & "' -> '" & preparedPriceRubStr & "') не является числом после подготовки"
                            End If
                        ElseIf tempDictAvail.Exists("RRP") Then
                            Dim rrpVal As Variant: rrpVal = tempDictAvail("RRP")
                            Debug.Print "    RRP (сырое, используется как RUB цена): '" & rrpVal & "'"
                            Dim preparedRrpStr As String: preparedRrpStr = Replace(CStr(rrpVal), ".", Application.DecimalSeparator)
                            If IsNumeric(preparedRrpStr) Then
                                Dim numericRRP As Double: numericRRP = CDbl(preparedRrpStr)
                                If numericRRP > 0 Then
                                    itemPriceRUB = numericRRP
                                    Debug.Print "      RRP (число, используется как RUB цена): " & itemPriceRUB
                                Else: Debug.Print "      RRP (" & numericRRP & ") <= 0"
                                End If
                            Else: Debug.Print "      RRP ('" & rrpVal & "' -> '" & preparedRrpStr & "') не является числом после подготовки"
                            End If
                        Else
                             Debug.Print "    Ключи 'PriceClientRUB' и 'RRP' ОТСУТСТВУЮТ в availApiDetails для '" & articulForApi & "'"
                        End If
                    Else
                        Debug.Print "    Элемент для '" & articulForApi & "' в availApiDetails НЕ ЯВЛЯЕТСЯ СЛОВАРЕМ. Тип: " & TypeName(tempDictAvail)
                        itemStock = "Ошибка данных (avail)"
                    End If
                Else: Debug.Print "  НЕ найден в availApiDetails для артикула '" & articulForApi & "'"
                End If
            Else: Debug.Print "  НЕ найден в itemApiDetails для артикула '" & articulForApi & "'"
            End If

            Debug.Print "  Запись в ячейки (строка " & currentRow & "): Name='" & itemName & "', Stock='" & itemStock & "', USD='" & itemPriceUSD & "', RUB='" & itemPriceRUB & "'"
            ws.Cells(currentRow, MERLION_COL_API_NAME).Value = itemName
            ws.Cells(currentRow, MERLION_COL_API_STOCK).Value = itemStock
            ws.Cells(currentRow, MERLION_COL_API_PRICE_USD).Value = itemPriceUSD
            ws.Cells(currentRow, MERLION_COL_API_PRICE_RUB).Value = itemPriceRUB
            ' Столбец AN не заполняется данными API
            
            totalProcessedArticuls = totalProcessedArticuls + 1
        Else
             If currentRow >= MERLION_DATA_START_ROW Then
                ws.Range(MERLION_COL_API_ARTICUL & currentRow & ":" & MERLION_COL_API_PRICE_RUB & currentRow).ClearContents ' Очищаем только AH-AM
             End If
        End If
         If currentRow Mod 20 = 0 Then Application.StatusBar = "Merlion: Обработано " & totalProcessedArticuls & "/" & totalValidArticulsToProcess
    Next currentRow

    MsgBox "Merlion: Обработка данных завершена! Обработано ВАЛИДНЫХ артикулов: " & totalProcessedArticuls, vbInformation
    Debug.Print "--- ProcessMerlionDataFromAPI (Merlion mlservice2) ЗАВЕРШЕН ---"

CleanUp_Merlion:
    Application.StatusBar = False
    Application.ScreenUpdating = True
    Application.Calculation = xlCalculationAutomatic
    Application.EnableEvents = True
    Set ws = Nothing
    If Not http Is Nothing Then Set http = Nothing
    Set itemApiDetails = Nothing
    Set availApiDetails = Nothing
    Exit Sub

GlobalErrorHandler_Merlion:
    Application.StatusBar = False
    Dim errNum As Long, errDesc As String, errLineVal As Long
    errNum = Err.Number: errDesc = Err.Description: errLineVal = Erl
    Dim errorRowInfo As String
    If currentRow > 0 Then
        errorRowInfo = "Обрабатывалась строка Excel (чтение/запись): " & currentRow
    Else
        errorRowInfo = "Ошибка до начала обработки строк Excel."
    End If
    MsgBox "Произошла ошибка VBA (Merlion): " & errNum & vbCrLf & errDesc & vbCrLf & "На строке VBA (Erl): " & errLineVal & vbCrLf & errorRowInfo, vbCritical
    Debug.Print "КРИТИЧЕСКАЯ ОШИБКА VBA (Merlion): " & errNum & " - " & errDesc & " на строке " & Erl & " (" & errorRowInfo & ")"
    Resume CleanUp_Merlion
End Sub

Private Sub Merlion_GetItem_API(ByVal singleArticul As String, ByRef itemResultsDict As Object, ByRef httpObject As Object)
    Dim xmlResponseDoc As Object, soapRequestBody As String, base64Credentials As String
    Dim currentStep As String
    
    On Error GoTo ErrorHandler_Merlion_GetItem_API_Main
    currentStep = "Начало Merlion_GetItem_API"

    If Trim(singleArticul) = "" Then
        Debug.Print "Merlion_GetItem_API: Пустой артикул, выход."
        Exit Sub
    End If
    Debug.Print "Merlion_GetItem_API: Артикул для запроса: " & singleArticul

    currentStep = "Проверка httpObject"
    If httpObject Is Nothing Then Set httpObject = CreateObject("WinHttp.WinHttpRequest.5.1")
    
    currentStep = "Создание xmlResponseDoc"
    Set xmlResponseDoc = CreateObject("MSXML2.DOMDocument.6.0")
    
    currentStep = "Настройка xmlResponseDoc"
    With xmlResponseDoc
        .async = False
        .SetProperty "SelectionLanguage", "XPath"
        .SetProperty "SelectionNamespaces", "xmlns:soapenv='http://schemas.xmlsoap.org/soap/envelope/' xmlns:mer='" & Replace(Replace(MER_NS_MERLION, "xmlns:mer='", ""), "'", "") & "'"
    End With
    
    currentStep = "Кодирование base64Credentials"
    base64Credentials = EncodeBase64_Merlion(MERLION_LOGIN & ":" & MERLION_PASSWORD)

    currentStep = "Формирование soapRequestBody для getItems"
    soapRequestBody = _
        "<soapenv:Envelope " & SOAPENV_NS_MERLION & " " & MER_NS_MERLION & ">" & _
        "   <soapenv:Header/>" & _
        "   <soapenv:Body>" & _
        "      <mer:getItems>" & _
        "         <mer:login>" & EncodeXMLText_Merlion(MERLION_LOGIN) & "</mer:login>" & _
        "         <mer:password>" & EncodeXMLText_Merlion(MERLION_PASSWORD) & "</mer:password>" & _
        "         <mer:item_id>" & EncodeXMLText_Merlion(singleArticul) & "</mer:item_id>" & _
        "      </mer:getItems>" & _
        "   </soapenv:Body>" & _
        "</soapenv:Envelope>"
    Debug.Print "Merlion_GetItem_API: Запрос для '" & singleArticul & "' (шаг: " & currentStep & "): " & Left(soapRequestBody, 300) & "..."

    With httpObject
        currentStep = "httpObject.Open для getItems"
        .Open "POST", MERLION_ENDPOINT_URL, False
        currentStep = "httpObject.setRequestHeader Content-Type для getItems"
        .setRequestHeader "Content-Type", "text/xml;charset=UTF-8"
        currentStep = "httpObject.setRequestHeader Authorization для getItems"
        .setRequestHeader "Authorization", "Basic " & base64Credentials
        currentStep = "httpObject.send для getItems"
        .send soapRequestBody
        
        currentStep = "После .send, перед получением статуса для getItems"
        Debug.Print "Merlion_GetItem_API: Статус ответа для '" & singleArticul & "' (шаг: " & currentStep & "): " & .Status

        If .Status = 200 Then
            currentStep = "Получение responseText для getItems"
            Dim responseTextVal As String
            responseTextVal = .responseText
            Debug.Print "Merlion_GetItem_API: ПОЛНЫЙ XML ОТВЕТ для '" & singleArticul & "' (шаг: " & currentStep & "): " & Left(responseTextVal, 2000)

            currentStep = "xmlResponseDoc.LoadXML для getItems"
            If xmlResponseDoc.LoadXML(responseTextVal) Then
                currentStep = "SelectSingleNode для getItems"
                Dim xPathQuery As String
                xPathQuery = "//*[local-name()='Envelope']/*[local-name()='Body']/*[local-name()='getItemsResponse']/*[local-name()='getItemsResult']/*[local-name()='item']"
                Debug.Print "Merlion_GetItem_API: XPath (local-name) для '" & singleArticul & "' (шаг: " & currentStep & "): " & xPathQuery
                
                Dim itemNode As Object, childDetailNode As Object, respName As String
                Dim nodesList As Object
                Set nodesList = xmlResponseDoc.SelectNodes(xPathQuery)
                Set itemNode = Nothing

                If Not nodesList Is Nothing Then
                    If nodesList.Length > 0 Then
                        Debug.Print "Merlion_GetItem_API: Найдено " & nodesList.Length & " <item> узлов для getItems."
                        For Each itemNode In nodesList
                            Dim noNode As Object
                            Set noNode = itemNode.SelectSingleNode("*[local-name()='No']")
                            If Not noNode Is Nothing Then
                                If Trim(noNode.Text) = Trim(singleArticul) Then
                                    Debug.Print "Merlion_GetItem_API: Найден СООТВЕТСТВУЮЩИЙ <item> для артикула '" & singleArticul & "' (getItems)."
                                    Exit For
                                End If
                            End If
                            Set itemNode = Nothing
                        Next itemNode
                    Else
                        Debug.Print "Merlion_GetItem_API: Список <item> узлов пуст для getItems."
                    End If
                Else
                    Debug.Print "Merlion_GetItem_API: SelectNodes вернул Nothing для <item> узлов (getItems)."
                End If
                                
                Dim tempDict As Object: Set tempDict = CreateObject("Scripting.Dictionary")
                If Not itemNode Is Nothing Then
                    Debug.Print "Merlion_GetItem_API: Узел для артикула '" & singleArticul & "' НАЙДЕН и ПОДТВЕРЖДЕН в XML (getItems)."
                    respName = "N/A (Имя не найдено в узле)"
                    For Each childDetailNode In itemNode.ChildNodes
                        If childDetailNode.BaseName = "Name" Then
                            respName = childDetailNode.Text
                            Exit For
                        End If
                    Next childDetailNode
                    tempDict("Name") = respName
                Else
                    Debug.Print "Merlion_GetItem_API: Артикул '" & singleArticul & "' не найден или не подтвержден в XML-ответе (getItems)."
                    tempDict("Name") = "Товар не найден (API getItems)"
                End If
                If itemResultsDict.Exists(Trim(singleArticul)) Then itemResultsDict(Trim(singleArticul)) = tempDict Else itemResultsDict.Add Trim(singleArticul), tempDict
            Else
                Debug.Print "Merlion_GetItem_API: Ошибка парсинга XML для '" & singleArticul & "' (шаг: " & currentStep & "). Причина: " & xmlResponseDoc.ParseError.reason
                Dim errDictParse As Object: Set errDictParse = CreateObject("Scripting.Dictionary")
                errDictParse("Name") = "Ошибка парсинга XML (items)"
                If itemResultsDict.Exists(Trim(singleArticul)) Then itemResultsDict(Trim(singleArticul)) = errDictParse Else itemResultsDict.Add Trim(singleArticul), errDictParse
            End If
        Else
            Debug.Print "Merlion_GetItem_API: Ошибка HTTP " & .Status & " для '" & singleArticul & "' (шаг: " & currentStep & "). " & .statusText
            Dim errDictHttp As Object: Set errDictHttp = CreateObject("Scripting.Dictionary")
            errDictHttp("Name") = "Ошибка API " & .Status & " (items)"
            If itemResultsDict.Exists(Trim(singleArticul)) Then itemResultsDict(Trim(singleArticul)) = errDictHttp Else itemResultsDict.Add Trim(singleArticul), errDictHttp
        End If
    End With
CleanExit_Merlion_GetItem_API:
    Set xmlResponseDoc = Nothing
    Exit Sub
ErrorHandler_Merlion_GetItem_API_Main:
    Debug.Print "КРИТИЧЕСКАЯ ОШИБКА в Merlion_GetItem_API для '" & singleArticul & "' на шаге '" & currentStep & "' (Строка Erl: " & Erl & "): " & Err.Number & " - " & Err.Description
    Dim errDictCrit As Object: Set errDictCrit = CreateObject("Scripting.Dictionary")
    errDictCrit("Name") = "КРИТ.ОШИБКА (items)"
    If Trim(singleArticul) <> "" Then
        If itemResultsDict.Exists(Trim(singleArticul)) Then itemResultsDict(Trim(singleArticul)) = errDictCrit Else itemResultsDict.Add Trim(singleArticul), errDictCrit
    End If
    Resume CleanExit_Merlion_GetItem_API
End Sub

Private Sub Merlion_GetItemAvail_API(ByVal singleArticul As String, ByRef availResultsDict As Object, ByRef httpObject As Object)
    Dim xmlResponseDoc As Object, soapRequestBody As String, base64Credentials As String, shipmentDateStr As String
    Dim currentStep As String

    On Error GoTo ErrorHandler_Merlion_GetItemAvail_API_Main
    currentStep = "Начало Merlion_GetItemAvail_API"

    If Trim(singleArticul) = "" Then
        Debug.Print "Merlion_GetItemAvail_API: Пустой артикул, выход."
        Exit Sub
    End If
    Debug.Print "Merlion_GetItemAvail_API: Артикул для запроса: " & singleArticul

    currentStep = "Проверка httpObject"
    If httpObject Is Nothing Then Set httpObject = CreateObject("WinHttp.WinHttpRequest.5.1")
    
    currentStep = "Создание xmlResponseDoc"
    Set xmlResponseDoc = CreateObject("MSXML2.DOMDocument.6.0")
    
    currentStep = "Настройка xmlResponseDoc"
    With xmlResponseDoc
        .async = False
        .SetProperty "SelectionLanguage", "XPath"
        .SetProperty "SelectionNamespaces", "xmlns:soapenv='http://schemas.xmlsoap.org/soap/envelope/' xmlns:mer='" & Replace(Replace(MER_NS_MERLION, "xmlns:mer='", ""), "'", "") & "'"

    shipmentDateStr = Format(Date, "YYYY-MM-DD")

    currentStep = "Формирование soapRequestBody для getItemsAvail"
    soapRequestBody = _
        "<soapenv:Envelope " & SOAPENV_NS_MERLION & " " & MER_NS_MERLION & ">" & _

        "         <mer:password>" & EncodeXMLText_Merlion(MERLION_PASSWORD) & "</mer:password>" & _
        "         <mer:item_id>" & EncodeXMLText_Merlion(singleArticul) & "</mer:item_id>" & _
        "         <mer:shipment_method>" & EncodeXMLText_Merlion(MERLION_SHIPMENT_METHOD) & "</mer:shipment_method>" & _
        "         <mer:shipment_date>" & shipmentDateStr & "</mer:shipment_date>" & _

        "</soapenv:Envelope>"
    Debug.Print "Merlion_GetItemAvail_API: Запрос для '" & singleArticul & "' (шаг: " & currentStep & "): " & Left(soapRequestBody, 350) & "..."

    With httpObject
        currentStep = "httpObject.Open для getItemsAvail"
        .Open "POST", MERLION_ENDPOINT_URL, False
        currentStep = "httpObject.setRequestHeader Content-Type для getItemsAvail"
        .setRequestHeader "Content-Type", "text/xml;charset=UTF-8"

        .setRequestHeader "Authorization", "Basic " & base64Credentials

        Debug.Print "Merlion_GetItemAvail_API: Статус ответа для '" & singleArticul & "' (шаг: " & currentStep & "): " & .Status

        If .Status = 200 Then
            currentStep = "Получение responseText для getItemsAvail"
            Dim responseTextVal As String
            responseTextVal = .responseText
            Debug.Print "Merlion_GetItemAvail_API: ПОЛНЫЙ XML ОТВЕТ для '" & singleArticul & "' (шаг: " & currentStep & "): " & Left(responseTextVal, 2000)

            currentStep = "xmlResponseDoc.LoadXML для getItemsAvail"
            If xmlResponseDoc.LoadXML(responseTextVal) Then

                Dim itemNode As Object, childDetailNode As Object
                Dim respStock As String, respPriceUSD As String, respPriceRUB As String, respRRP As String
                Dim xPathQuery As String
                
                xPathQuery = "//*[local-name()='Envelope']/*[local-name()='Body']/*[local-name()='getItemsAvailResponse']/*[local-name()='getItemsAvailResult']/*[local-name()='item']"
                Debug.Print "Merlion_GetItemAvail_API: XPath (local-name) для '" & singleArticul & "' (шаг: " & currentStep & "): " & xPathQuery
                
                Dim nodesList As Object
                Set nodesList = xmlResponseDoc.SelectNodes(xPathQuery)
                Set itemNode = Nothing

                If Not nodesList Is Nothing Then
                    If nodesList.Length > 0 Then
                         Debug.Print "Merlion_GetItemAvail_API: Найдено " & nodesList.Length & " <item> узлов для getItemsAvail."
                        For Each itemNode In nodesList
                            Dim noNode As Object
                            Set noNode = itemNode.SelectSingleNode("*[local-name()='No']")
                            If Not noNode Is Nothing Then
                                If Trim(noNode.Text) = Trim(singleArticul) Then
                                     Debug.Print "Merlion_GetItemAvail_API: Найден СООТВЕТСТВУЮЩИЙ <item> для артикула '" & singleArticul & "' (getItemsAvail)."
                                    Exit For
                                End If
                            End If
                            Set itemNode = Nothing
                        Next itemNode
                    Else
                        Debug.Print "Merlion_GetItemAvail_API: Список <item> узлов пуст для getItemsAvail."
                    End If
                Else
                     Debug.Print "Merlion_GetItemAvail_API: SelectNodes вернул Nothing для <item> узлов (getItemsAvail)."
                End If
                
                Dim tempDict As Object: Set tempDict = CreateObject("Scripting.Dictionary")
                respStock = vbNullString: respPriceUSD = "0": respPriceRUB = "0": respRRP = "0"

                If Not itemNode Is Nothing Then
                    Debug.Print "Merlion_GetItemAvail_API: Узел для артикула '" & singleArticul & "' НАЙДЕН и ПОДТВЕРЖДЕН в XML (getItemsAvail)."
                    For Each childDetailNode In itemNode.ChildNodes
                        Select Case childDetailNode.BaseName
                            Case "AvailableClient": respStock = childDetailNode.Text
                            Case "PriceClient": respPriceUSD = childDetailNode.Text
                            Case "PriceClientRUB": respPriceRUB = childDetailNode.Text
                            Case "RRP": respRRP = childDetailNode.Text
                        End Select
                    Next childDetailNode
                Else
                    Debug.Print "Merlion_GetItemAvail_API: Артикул '" & singleArticul & "' не найден или не подтвержден в XML-ответе (getItemsAvail)."
                End If
                tempDict("Stock") = respStock
                tempDict("PriceUSD") = respPriceUSD
                tempDict("PriceClientRUB") = respPriceRUB
                tempDict("RRP") = respRRP
                If availResultsDict.Exists(Trim(singleArticul)) Then availResultsDict(Trim(singleArticul)) = tempDict Else availResultsDict.Add Trim(singleArticul), tempDict
            Else
                Debug.Print "Merlion_GetItemAvail_API: Ошибка парсинга XML для '" & singleArticul & "' (шаг: " & currentStep & "). Причина: " & xmlResponseDoc.ParseError.reason
                Dim errDictParse As Object: Set errDictParse = CreateObject("Scripting.Dictionary")
                errDictParse("Stock") = vbNullString: errDictParse("PriceUSD") = "0": errDictParse("PriceClientRUB") = "0": errDictParse("RRP") = "0"
                If availResultsDict.Exists(Trim(singleArticul)) Then availResultsDict(Trim(singleArticul)) = errDictParse Else availResultsDict.Add Trim(singleArticul), errDictParse
            End If
        Else
            Debug.Print "Merlion_GetItemAvail_API: Ошибка HTTP " & httpObject.Status & " для '" & singleArticul & "' (шаг: " & currentStep & "). " & httpObject.statusText
            Dim errDictHttp As Object: Set errDictHttp = CreateObject("Scripting.Dictionary")
            errDictHttp("Stock") = vbNullString: errDictHttp("PriceUSD") = "0": errDictHttp("PriceClientRUB") = "0": errDictHttp("RRP") = "0"
            If availResultsDict.Exists(Trim(singleArticul)) Then availResultsDict(Trim(singleArticul)) = errDictHttp Else availResultsDict.Add Trim(singleArticul), errDictHttp
        End If
    End With
CleanExit_Merlion_GetItemAvail_API:
    Set xmlResponseDoc = Nothing
    Exit Sub
ErrorHandler_Merlion_GetItemAvail_API_Main:
    Debug.Print "КРИТИЧЕСКАЯ ОШИБКА в Merlion_GetItemAvail_API для '" & singleArticul & "' на шаге '" & currentStep & "' (Строка Erl: " & Erl & "): " & Err.Number & " - " & Err.Description
    Dim errDictCritical As Object: Set errDictCritical = CreateObject("Scripting.Dictionary")
    errDictCritical("Stock") = "ОШИБКА": errDictCritical("PriceUSD") = "0": errDictCritical("PriceClientRUB") = "0": errDictCritical("RRP") = "0"
    If Trim(singleArticul) <> "" Then
        If availResultsDict.Exists(Trim(singleArticul)) Then availResultsDict(Trim(singleArticul)) = errDictCritical Else availResultsDict.Add Trim(singleArticul), errDictCritical
    End If
    Resume CleanExit_Merlion_GetItemAvail_API
End Sub

Private Function EncodeBase64_Merlion(sText As String) As String
    Dim arrData() As Byte, objXML As Object, objNode As Object
    On Error GoTo HELL_EncodeBase64_Merlion
    arrData = StrConv(sText, vbFromUnicode)
    Set objXML = CreateObject("MSXML2.DOMDocument.6.0")
    Set objNode = objXML.createElement("b64")
    objNode.DataType = "bin.base64": objNode.nodeTypedValue = arrData
    EncodeBase64_Merlion = objNode.Text
    Set objNode = Nothing: Set objXML = Nothing
    Exit Function
HELL_EncodeBase64_Merlion:
    EncodeBase64_Merlion = ""
    Debug.Print "Ошибка в EncodeBase64_Merlion: " & Err.Number & " - " & Err.Description
End Function

Private Function EncodeXMLText_Merlion(ByVal textToEncode As String) As String
    If textToEncode = "" Then
        EncodeXMLText_Merlion = ""
        Exit Function
    End If
    Dim tempString As String
    tempString = textToEncode
    tempString = Replace(tempString, "&", "&")
    tempString = Replace(tempString, "<", "<")
    tempString = Replace(tempString, ">", ">")
    tempString = Replace(tempString, "'", "'")
    tempString = Replace(tempString, """", """")
    EncodeXMLText_Merlion = tempString
End Function
